async function fetchAirtable(table) {
  const rows = [];
  let offset = '';
  do {
    const url = `/api/read?table=${encodeURIComponent(table)}${offset?'&offset='+offset:''}`;
    const r   = await fetch(url);
    const d   = await r.json();
    (d.records||[]).forEach(rec => rows.push(rec.fields));
    offset = d.offset||'';
  } while (offset);
  return rows;
}

// ─── САЙДБАР ────────────────────────────────────────────────
function buildSidebar() {
  const aEl = document.getElementById('sidebar-mods-a');
  const bEl = document.getElementById('sidebar-mods-b');
  aEl.innerHTML = ''; bEl.innerHTML = '';
  modules.forEach((m,i) => {
    const el = document.createElement('div');
    el.className = 'mod-item';
    el.id = 'sidebar-mod-' + i;
    el.onclick = () => showModule(i);
    el.innerHTML = `<div class="mod-num-badge">${m.id}</div><div class="mod-item-info"><div class="mod-item-name">${m.title}</div><div class="mod-item-dur">${m.duration}${m.pro?' <span class="pro-badge">Потрібен Pro</span>':''}</div></div>`;
    if (m.track==='a') aEl.appendChild(el);
    else bEl.appendChild(el);
  });
}

// ─── ПОКАЗ МОДУЛЯ ───────────────────────────────────────────
function showModule(idx) {
  currentMod = idx;
  const m = modules[idx];
  document.querySelectorAll('.mod-item').forEach(el=>el.classList.remove('active'));
  const sEl = document.getElementById('sidebar-mod-'+idx);
  if (sEl) sEl.classList.add('active');
  document.getElementById('nav-title').textContent = m.title;
  updateProgress();
  const area = document.getElementById('content-area');
  area.innerHTML = `
    <div class="module-header">
      <div class="mod-tag">${m.tag}</div>
      <div class="mod-title">${m.title}</div>
      <div class="mod-meta">
        <div class="meta-item"><div class="meta-dot"></div>${m.lessons} уроки</div>
        <div class="meta-item"><div class="meta-dot"></div>${m.duration}</div>
        ${m.pro?'<div class="meta-item"><div class="meta-dot" style="background:#4C88FF"></div>Потрібен Pro</div>':''}
      </div>
      <div class="objective-box"><div class="obj-label">РЕЗУЛЬТАТ МОДУЛЯ</div><div class="obj-text">${m.objective}</div></div>
    </div>
    ${m.content}
    <div class="mod-nav">
      ${idx>0?`<button class="nav-btn" onclick="showModule(${idx-1})">← Попередній</button>`:'<div></div>'}
      ${idx<modules.length-1?`<button class="nav-btn primary" onclick="showModule(${idx+1})">Наступний →</button>`:'<div class="nav-btn primary" style="opacity:0.5;pointer-events:none">Курс завершено 🎉</div>'}
    </div>`;
  area.scrollTop=0; window.scrollTo(0,0);
  restoreDoneState(idx);
}

function updateProgress() {
  const done = modules.filter(m=>doneModules.has(m.id)).length;
  const pct  = modules.length ? (done/modules.length*100) : 0;
  document.getElementById('nav-progress-text').textContent = done+' / '+modules.length;
  document.getElementById('pbar-fill').style.width = pct+'%';
}

// ─── УРОКИ ──────────────────────────────────────────────────
function toggleLesson(modId, lesId, card) {
  const body   = document.getElementById('lb-'+modId+'-'+lesId);
  const isOpen = body.classList.contains('open');
  modules[modId] && Array.from({length:modules[modId].lessons},(_,i)=>{
    const b=document.getElementById('lb-'+modId+'-'+i);
    const c=document.getElementById('lc-'+modId+'-'+i);
    if(b) b.classList.remove('open');
    if(c&&!c.classList.contains('done')) c.classList.remove('active');
  });
  if(!isOpen) {
    body.classList.add('open');
    if(!card.classList.contains('done')) card.classList.add('active');
  }
}

async function markDone(modId, lesId, e) {
  e.stopPropagation();
  const key    = modId+'-'+lesId;
  doneMap[key] = true;
  const card   = document.getElementById('lc-'+key);
  const status = document.getElementById('ls-'+key);
  const body   = document.getElementById('lb-'+key);
  card.classList.remove('active'); card.classList.add('done');
  status.textContent = '✓';
  body.classList.remove('open');
  await track('lesson', { modId, modName: modules[modId].title, lesId });
  const m       = modules[modId];
  const allDone = Array.from({length:m.lessons},(_,i)=>doneMap[modId+'-'+i]).every(Boolean);
  if(allDone) {
    doneModules.add(modId);
    const sEl = document.getElementById('sidebar-mod-'+modId);
    if(sEl) sEl.classList.add('done');
    updateProgress();
  }
}

function answerQuiz(el, type, modId, qIdx) {
  const group = el.closest('.quiz-opts');
  if(group.querySelector('.answered')) return;
  group.querySelectorAll('.quiz-opt').forEach(opt=>{ opt.classList.add('answered'); opt.style.pointerEvents='none'; });
  if(type==='correct') { el.classList.add('correct'); el.innerHTML+=' ✓'; }
  else { el.classList.add('wrong'); el.innerHTML+=' ✗'; const c=group.querySelector('[onclick*="correct"]'); if(c) c.classList.add('correct'); }
  track('quiz', { modId: modId||currentMod, qIdx: qIdx||1, answer: el.textContent.replace(' ✗','').replace(' ✓','').trim(), correct: type==='correct' });
}

  const m = modules[modId];

}
